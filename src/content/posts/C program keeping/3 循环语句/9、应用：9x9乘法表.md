### 9.2、`while`循环实现

```c
//
// Created by wangmingdong on 2023/6/4.
//

#include "stdio.h"

int main() {

    int i = 1;
    while (i <= 9) {  // 控制外层循环，主要用来换行

        int j = 1;
        while (j <= i) {  // 控制内层循环，主要用来在1行中打印多个*
            printf("%d * %d = %-2d   ", j, i, j * i);
            j++;
        }
        printf("\n");

        i++;
    }

    return 0;
}
```

### 9.3、`do-while`循环实现

```c
//
// Created by wangmingdong on 2023/6/4.
//

#include "stdio.h"

int main() {

    int i = 1;
    do {  // 控制外层循环，主要用来换行

        int j = 1;
        do {  // 控制内层循环，主要用来在1行中打印多个*
            printf("%d * %d = %-2d   ", j, i, j * i);
            j++;
        } while (j <= i);
        printf("\n");

        i++;
    } while (i <= 9);

    return 0;
}
```

### 9.4、`for`循环实现

```c
//
// Created by wangmingdong on 2023/6/4.
//

#include "stdio.h"

int main() {

    for (int i = 1; i <= 9; i++) {  // 控制外层循环，主要用来换行
        for (int j = 1; j <= i; j++) {  // 控制内层循环，主要用来在1行中打印多个*
            printf("%d * %d = %-2d   ", j, i, j * i);
        }
        printf("\n");
    }

    return 0;
}
```
